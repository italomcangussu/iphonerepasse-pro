# Reservation Deposit PDV Finance Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Paid stock reservations enter Finance as advances, can be sold through PDV without duplicate cash-in, and can be canceled with an explicit refund-or-retain decision.

**Architecture:** Keep all money-moving reservation operations inside Supabase RPCs and expose them through DataProvider actions. Use a small PDV reserved-sale prefill module so the normal product picker remains safe while `Vender reservado` intentionally loads reserved stock into the sale flow. Extend sale payment metadata so reservation deposits count toward PDV balance and sale history while the sale RPC skips duplicate Finance transactions.

**Tech Stack:** React 19, Vite, TypeScript, Vitest/jsdom, Supabase Postgres migrations/RPCs, existing DataProvider and PDV RPC architecture.

## Global Constraints

- Do not touch n8n files for this feature.
- The worktree currently has unrelated n8n changes; never stage or commit them with this feature.
- Use TDD for every behavior change: write a failing test, run it red, implement, run it green.
- Add a new Supabase migration instead of editing historical migrations.
- Keep the regular PDV product picker limited to `StockStatus.AVAILABLE`.
- `Dinheiro` reservation deposits go to `Cofre`; `Pix`, `Cartão`, `Cartão Débito`, and `Outro` go to `Conta Bancária`.
- Reservation deposit sale payments use `source = 'reservation_deposit'`.
- The sale RPC must persist reservation-deposit payment rows but skip creating Finance `IN` transactions for them.
- Text shown to users remains Brazilian Portuguese.
- Verification gates before completion: targeted tests, `npm run test:run`, `npm run lint`, and `npm run typecheck`.

- [ ] **Step 5: Add `release_stock_reservation` RPC**

Add:

```sql
create or replace function public.release_stock_reservation(
  p_stock_item_id text,
  p_refund_deposit boolean default false
)
returns public.stock_reservations
language plpgsql
security definer
set search_path = public
as $$
declare
  v_stock_item public.stock_items%rowtype;
  v_reservation public.stock_reservations%rowtype;
  v_deposit_transaction public.transactions%rowtype;
  v_refund_transaction_id text;
begin
  select * into v_stock_item
  from public.stock_items
  where id = p_stock_item_id
  for update;

  if not found then
    raise exception 'Aparelho não encontrado no estoque.';
  end if;

  select * into v_reservation
  from public.stock_reservations
  where stock_item_id = p_stock_item_id
    and status = 'active'
  for update;

  if not found then
    raise exception 'Reserva ativa não encontrada.';
  end if;

  if v_reservation.deposit_refund_transaction_id is not null then
    raise exception 'Sinal da reserva já foi estornado.';
  end if;

  if p_refund_deposit and coalesce(v_reservation.deposit_amount, 0) > 0 then
    if v_reservation.deposit_transaction_id is not null then
      select * into v_deposit_transaction
      from public.transactions
      where id = v_reservation.deposit_transaction_id
      for update;
    end if;

    v_refund_transaction_id := 'trx_' || replace(gen_random_uuid()::text, '-', '');

    insert into public.transactions (
      id, type, category, amount, date, description, account
    ) values (
      v_refund_transaction_id,
      'OUT',
      'Estorno de reserva',
      coalesce(v_reservation.deposit_amount, 0),
      now(),
      'Estorno de reserva - ' || v_reservation.customer_name || ' - ' || v_stock_item.model,
      coalesce(v_deposit_transaction.account, public.reservation_deposit_account(v_reservation.deposit_payment_method))
    );

    update public.stock_reservations
       set status = 'released',
           released_at = now(),
           deposit_refund_transaction_id = v_refund_transaction_id,
           deposit_refunded_at = now(),
           deposit_retained_at = null
     where id = v_reservation.id
     returning * into v_reservation;
  else
    update public.stock_reservations
       set status = 'released',
           released_at = now(),
           deposit_refund_transaction_id = null,
           deposit_refunded_at = null,
           deposit_retained_at = case
             when coalesce(deposit_amount, 0) > 0 then now()
             else null
           end
     where id = v_reservation.id
     returning * into v_reservation;
  end if;

  update public.stock_items
     set status = 'Disponível',
         updated_at = now()
   where id = p_stock_item_id;

  return v_reservation;
end;
$$;
```

Create `pages/pdv/pdvReservedPrefill.ts`:

```ts
import type { PaymentMethod } from '../../types';
import type { StorageLike } from './pdvDraft';

export const PDV_RESERVED_SALE_PREFILL_KEY = 'pdv:reserved-sale-prefill:v1';
const PDV_RESERVED_SALE_PREFILL_VERSION = 1;

export type ReservedSalePrefill = {
  stockItemId: string;
  reservationId: string;
  selectedStore: string;
  selectedClient?: string;
  depositPayment?: PaymentMethod;
};

type VersionedReservedSalePrefill = {
  version: typeof PDV_RESERVED_SALE_PREFILL_VERSION;
  prefill: ReservedSalePrefill;
};

const isRecord = (value: unknown): value is Record<string, unknown> =>
  typeof value === 'object' && value !== null;

const isValidPrefill = (value: unknown): value is ReservedSalePrefill =>
  isRecord(value) &&
  typeof value.stockItemId === 'string' &&
  value.stockItemId.trim().length > 0 &&
  typeof value.reservationId === 'string' &&
  value.reservationId.trim().length > 0 &&
  typeof value.selectedStore === 'string' &&
  value.selectedStore.trim().length > 0;

const isVersionedPrefill = (value: unknown): value is VersionedReservedSalePrefill =>
  isRecord(value) &&
  value.version === PDV_RESERVED_SALE_PREFILL_VERSION &&
  isValidPrefill(value.prefill);

export const readReservedSalePrefill = (storage: StorageLike): ReservedSalePrefill | null => {
  try {
    const raw = storage.getItem(PDV_RESERVED_SALE_PREFILL_KEY);
    if (!raw) return null;
    const parsed: unknown = JSON.parse(raw);
    return isVersionedPrefill(parsed) ? parsed.prefill : null;
  } catch {
    return null;
  }
};

export const writeReservedSalePrefill = (storage: StorageLike, prefill: ReservedSalePrefill): void => {
  storage.setItem(PDV_RESERVED_SALE_PREFILL_KEY, JSON.stringify({
    version: PDV_RESERVED_SALE_PREFILL_VERSION,
    prefill
  }));
};

export const clearReservedSalePrefill = (storage: StorageLike): void => {
  storage.removeItem(PDV_RESERVED_SALE_PREFILL_KEY);
};
```
